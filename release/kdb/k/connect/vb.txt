there is a single function k(s,...)
string s is executed and applied to the rest.

k	vb	

int	long(integer,boolean,date)
float	double(single,currency,timestamp)
char	byte
symbol	string
null	empty(null)
(..;..)	array(..,..)

sample module:

 declare function k lib"k20"(s,paramarray a())

 sub main()
 k"\b s"			' stop on \
 r=k("{[i;f;s]\s}",2,2.5,"v")	' pass 3 args
 debug.print r
 end sub


^G immediate window

typename(x)	' type
1+ubound(x)	' count
