 with vb: shell("excel")
k as the universal joint

callback vb sub and function

sub s(x,y)
debug.print y
msgbox x
end sub

function f(x,y,z)
f=x+y+z
end function

sub main()
k "{f::2:x,y}",addressof f,3   ' register f for 3 args
k "{s::2:x,y}",addressof s,8+2 ' register s for 2 args
k "s[`boo;f[2;3;4]]"           ' call f;call s						
end sub
' subs are indicated with an 8+(number of args)
